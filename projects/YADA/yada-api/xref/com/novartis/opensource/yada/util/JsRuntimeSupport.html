15  package com.novartis.opensource.yada.util;
16  
17  import java.io.FileNotFoundException;
18  import java.io.IOException;
19  import java.io.InputStreamReader;
20  import java.net.URL;
21  
22  import org.mozilla.javascript.Context;
23  import org.mozilla.javascript.Function;
24  import org.mozilla.javascript.Scriptable;
25  import org.mozilla.javascript.ScriptableObject;
26  
27  /**
28   * Utility class/custom implementation of abstract {@link ScriptableObject} to enable use of 
29   * {@code require.js} to load 3rd party libs into js scope.
30   * 
31   * @author David Varon
32   * @since 6.1.0
33   * @see "http://requirejs.org"
34   */
35  public class JsRuntimeSupport extends ScriptableObject {
36  
37      private static final long serialVersionUID = 1L;
38      private static final boolean silent = false;
39  
40      /**
41       * Convenience method to display js content in std out.
42       * 
43       * @param cx the current context
44       * @param thisObj the current scope
45       * @param args arguments to pass
46       * @param funObj 
47       */
48      public static void print(Context cx, Scriptable thisObj, Object[] args, Function funObj) 
49      {
50        if (silent)
51          return;
52      }
53  
54      /**
55       * @param cx the current context
56       * @param thisObj the current scope
57       * @param args arguments to pass
58       * @param funObj
59       * @throws FileNotFoundException when the file can't be located
60       * @throws IOException when there is a problem reading the file
61       */
62      public static void load(Context cx, Scriptable thisObj, Object[] args, Function funObj) throws FileNotFoundException, IOException 
63      {
64        JsRuntimeSupport shell = (JsRuntimeSupport) getTopLevelScope(thisObj);
65        for (int i = 0; i < args.length; i++) 
66        {
67            shell.processSource(cx, Context.toString(args[i]));
68        }
69      }
70  
71      /**
72       * Utility method to do the actual file I/O
73       * @param cx current js context
74       * @param url location of resource
75       * @throws FileNotFoundException 
76       * @throws IOException
77       */
78      private void processSource(Context cx, String url) throws FileNotFoundException, IOException 
79      {
80        URL u = new URL(url);
81        String[] elements = url.split("/");  
82        String filename = elements[elements.length - 1];
83        cx.evaluateReader(this, new InputStreamReader(u.openStream()), filename, 0, null);
84      }
85  
86      @Override
87      public String getClassName() {
88        return "JsRuntimeSupport";
89      }
90  
91  }
