17  import java.io.BufferedReader;
18  import java.io.IOException;
19  import java.io.StringReader;
20  
21  import org.apache.log4j.Logger;

23  import com.novartis.opensource.yada.YADAQueryResult;

25  /**
26   * Primarily a tool to assist with query authoring, this response returns query results wrapped in a simple HTML table.
27   * @author David Varon
28   *
29   */
30  public class HTMLResponse extends AbstractResponse {
31  	/**
32  	 * Local logger handle
33  	 */
34  	@SuppressWarnings("unused")
35  	private static Logger l = Logger.getLogger(HTMLResponse.class);
36  	/**
37     * Ivar containing the result to be returned by this class's {@link #toString()} method
38     */
39    private StringBuffer buffer = new StringBuffer();
40  	
41  	
42  	@Override
43    public Response compose(YADAQueryResult[] yqrs) throws YADAConverterException, YADAResponseException 
44  	{
45  	  Response delimitedResponse = new DelimitedResponse();
46  	  String csv = delimitedResponse.compose(yqrs).toString();
47  
48  	  try(BufferedReader br = new BufferedReader(new StringReader(csv)))
49  	  {
50  	    String line = "";
51  	    int lineNum = 0;
52  	    int fields  = 0;
53  	    int hdrFields = 0;
54  	    String oTag = "<th>";
55  	    String cTag = "</th>";
56  	    String oRow = "<tr>";
57  	    String cRow = "</tr>\n";
58  	    this.buffer.append("<html>\n<head>\n\n</head>\n<body>\n");
59  	    this.buffer.append("<table border=\"1\" style=\"border-collapse:collapse;\">\n");
60  	    while((line = br.readLine()) != null)
61  	    {
62  	      fields = 0;
63  	      if(lineNum == 0)
64  	      {
65  	        this.buffer.append("<thead>\n");
66  	      }
67  	      else
68  	      {
69  	        oTag = "<td>"; 
70  	        cTag = "</td>";
71  	      }
72    	    this.buffer.append(oRow);
73  	      char[] p = line.toCharArray(); // the values passed in
74  	      boolean escape = false, delim = false, inField = false;
75  	      // "Z,Z",,,"2013-04-03","2013-04-04 20:12:10" 
76  	      for(int i=0;i<p.length;i++)
77  	      {
78  	        char c = p[i];
79  	        // slash is probably escape char
80  	        if(c == '/')
81  	        {
82  	          this.buffer.append(c); // add it
83  	          escape = true;  // set state
84  	        }
85  	        // quote could be opener, closer, or escaped
86  	        else if(c == '"')  
87  	        {
88  	          if(inField) // already in a field
89  	          {
90  	            if(escape) // an escaped quote
91  	            {
92  	              this.buffer.append(c); // add it
93  	              escape = false; // unset state
94  	            }
95  	            else
96  	            {
97  	              this.buffer.append(cTag); // in a field and not escaped so close it
98  	              fields++;          // increment fields counter;
99  	              inField = false;   // unset state
100 	              delim = true;      // pending delimiter set state
101 	            }
102 	          }
103 	          else
104 	          {
105 	            this.buffer.append(oTag); // not in a field, so open it
106 	            inField = true;    // set state
107 	          }
108 	        }
109 	        else if(c == ',')
110 	        {
111 	          if(inField)
112 	          {
113 	            this.buffer.append(c); // in a field, so add it
114 	          }
115 	          else
116 	          {
117 	            if(delim)
118 	            {
119 	              delim = false; // not in field, and field just closed
120 	            }
121 	            else
122 	            {
123 	              this.buffer.append(oTag+cTag); // not in a field, no pending delim, so this is empty field
124 	              fields++;
125 	            }
126 	          } 
127 	        }
128 	        else
129 	        {
130 	          this.buffer.append(c); // any other char
131 	          escape = false;
132 	        }
133 	      }
134 	      if(fields < hdrFields)
135 	      {
136 	        for(int i=0;i<hdrFields-fields;i++)
137 	        {
138 	          this.buffer.append(oTag+cTag);
139 	        }
140 	      }
141 	      this.buffer.append(cRow);
142         if(lineNum == 0)
143         {
144           this.buffer.append("</thead>\n<tbody>\n");
145           hdrFields = fields;
146           oTag = "<td>"; 
147           cTag = "</td>";
148         }
149         lineNum++;
150 	    }
151 	    this.buffer.append("</tbody>\n</table>\n</body>\n</html>");
152 	  }
153 	  catch(IOException e)
154 	  {
155 	    String msg = "There was a problem parsing the result.";
156 	    throw new YADAResponseException(msg, e);
157 	  }
158 	  return this;
159 	}

167 		return this.buffer.toString();
168 	}
