32  /**
33   * Facilitates comparison and evaluation of xml sources.
34   * @author David Varon
35   *
36   */
37  public class XmlUtils {
38  
39  		/**
40  		 * Returns {@code true} if {@code xml1} is equal to {@code xml2}
41  		 * @param xml1 element list to compare
42  		 * @param xml2 element list to compare
43  		 * @return {@code true} if {@code xml1} is equal to {@code xml2}
44  		 */
45  		public boolean compareElementLists(String xml1, String xml2)
46  		{
47  			String xmlStr1 = getNormalElementList(xml1);
48  			String xmlStr2 = getNormalElementList(xml2);
49  			return xmlStr1.equals(xmlStr2);
50  		}
51  		
52  		/**
53  		 * Transforms {@code xml} into a simple list of elements, using {@code $YADA_HOME/util/elementlist.xsl}
54  		 * @param xml xml string from which to extract element names
55  		 * @return a list of elements
56  		 */
57  		public String getNormalElementList(String xml)
58  		{
59  			String result = "";
60  			StringWriter output = new StringWriter();
61  			// 1. Instantiate a TransformerFactory.
62  			TransformerFactory tFactory =  TransformerFactory.newInstance();
63  
64  			// 2. Use the TransformerFactory to process the stylesheet Source and generate a Transformer.
65  			Transformer transformer;
66  			try {
67  				transformer = tFactory.newTransformer(new javax.xml.transform.stream.StreamSource(System.getenv("YADA_HOME")+"/util/elementlist.xsl"));
68  				// 3. Use the Transformer to transform an XML Source and send the output to a Result object.
69  				transformer.transform
70  			    	(new StreamSource(new StringReader(xml)), 
71  			    	 new StreamResult(output));
72  				result = output.getBuffer().toString();
73  			} catch (TransformerConfigurationException e) {
74  				e.printStackTrace();
75  			} catch (TransformerException e) {
76  				e.printStackTrace();
77  			}
78  			return result;
79  		}
80  		
81  		/**
82  		 * Returns the text content of {@code elementName} from {@code xml}
83  		 * @param xml xml string containing content
84  		 * @param elementName the element from which to get the content
85  		 * @return text content
86  		 */
87  		public String getContentByElementName(String xml, String elementName)
88  		{
89  			String result = "";
90  			String xpath = "//"+elementName;
91  			try
92  			{
93  				XPath xpathObj = XPathFactory.newInstance().newXPath();
94  				result = xpathObj.evaluate(xpath,new InputSource(new StringReader(xml)));
95  			}
96  			catch (XPathExpressionException e)
97  			{
98  				e.printStackTrace();
99  			}
100 			return result;
101 		}
