31  /**
32   * An internally used data structure encapsulating data and params associated to a {@link YADAQuery} via {@code qname},
33   * passed to the framework in a request.
34   *  
35   * @author David Varon
36   * @since 0.4.0.0
37   * @see JSONParams
38   */
39  public class JSONParamsEntry { 
40  	
41  	/**
42  	 * Local logger handle
43  	 */
44  	private static Logger l = Logger.getLogger(JSONParamsEntry.class);
45  	/**
46  	 * Internal structure to hold column names/data values passed in request config 
47  	 */
48  	private List<LinkedHashMap<String,String[]>> data;
49  	/**
50  	 * Request config in list form
51  	 */
52  	private List<YADAParam> params;
53  	/**
54  	 * Internal index of parameters mapped by name
55  	 */
56  	private Map<String,YADAParam> keys = new HashMap<>();
57  	/**
58  	 * Internal index of immutable parameters mapped by name 
59  	 */
60  	private Map<String,YADAParam> immutableKeys = new HashMap<>();
61  	
62  	/**
63  	 * The default constructor, sets the internal {@link #data} and {@link #params} structures.
64  	 */
65  	public JSONParamsEntry() {
66  		setData(new ArrayList<LinkedHashMap<String,String[]>>());
67  		setParams(new ArrayList<YADAParam>());
68  	}
69  	
70  	/**
71  	 * Mutator for variable
72  	 * @param data the data map to set or use as replacement
73  	 */
74  	private void setData(ArrayList<LinkedHashMap<String,String[]>> data)
75  	{
76  		this.data = data;
77  	}
78  
79  	/**
80  	 * Mutator for variable, plus, if {@code params} size > 0, calls {@link #setKeys()} 
81  	 * to populate internal indices.
82  	 * @param params list of YADA request parameters
83  	 */
84  	private void setParams(ArrayList<YADAParam> params)
85  	{
86  		this.params = params;
87  		if(params.size() > 0) 
88  		{
89  			setKeys();
90  		}
91  	}
92  	
93  	/**
94  	 * Iterates over {@link #params} mapping parameter objects to parameter names
95  	 */
96  	private void setKeys() 
97  	{
98  		for(YADAParam param : this.params)
99  		{
100 			this.keys.put(param.getName(), param);
101 			if(param.getRule() != YADAParam.OVERRIDEABLE)
102 				this.immutableKeys.put(param.getName(), param);
103 		}
104 	}

106 	/**
107 	 * Creates an instance with populated {@link #data} and {@link #params} objects.
108 	 * @param jobj a JSONObject containing data and parameters
109 	 * @throws YADAQueryConfigurationException when {@code jobj} is malformed, or otherwise can't be converted into an entry 
110 	 */
111 	public JSONParamsEntry(JSONObject jobj) throws YADAQueryConfigurationException
112 	{
113 		this();
114 	
115 		try
116 		{
117 			for(String frag : YADAUtils.PARAM_FRAGS)
118 			{
119 				String key = YADARequest.getParamKeyVal("PS_"+frag);
120 				String val = YADARequest.getParamValueForKey(jobj,frag);
121 				if(val != null)
122 				{
123 					YADAParam param = new YADAParam(key,val,YADAParam.QUERY,YADAParam.OVERRIDEABLE);
124 					addParam(param);
125 				}
126 			}
127 			
128 			JSONArray  rows  = jobj.getJSONArray(YADARequest.JSON_KEY_DATA); // query DATA
129 			// multiple rows of data
130 			for (int j = 0; j < rows.length(); j++) 
131 			{
132 				JSONObject  row  = rows.getJSONObject(j); // row
133 				Iterator<?> iter = row.keys();
134 				LinkedHashMap<String,String[]> dataForRow = new LinkedHashMap<>();
135 				while (iter.hasNext()) 
136 				{
137 					String   column = ((String) iter.next()).toUpperCase(); //TODO this likely affects case-sensitivity
138 					String[] value  = null; 
139 					
140 					// JSONArrays can be passed in as values
141 					JSONArray valIsArray;
142 					try
143 					{
144 						valIsArray = row.getJSONArray(column);
145 						value = new String[valIsArray.length()];
146 						for (int k=0;k<valIsArray.length();k++)
147 						{
148 							value[k]  = valIsArray.getString(k);
149 						}
150 						l.debug("JSONArray passed in is now ["+value+"]");
151 					}
152 					catch(JSONException e)
153 					{
154 						// value passed in was just a string
155 						value = new String[] {String.valueOf(row.get(column))};
156 					}
157 					dataForRow.put(column, value);
158 				}
159 				// store the col/val hash in the arraylist for the qname
160 				addData(dataForRow); 
161 			}
162 		}
163 		catch(JSONException e)
164 		{
165 			String msg = "Unable to create JSONParamsEntry object from supplied parameters.";
166 			throw new YADAQueryConfigurationException(msg,e);
167 		}
168 	}

186 	/**
187 	 * Adds {@code row} to the entry's data list.
188 	 * @param row the data row to add to the existing data structure in the entry.
189 	 */
190 	public void addData(JSONObject row) {
191 		LinkedHashMap<String,String[]> d = new LinkedHashMap<>();
192 		String[] k = JSONObject.getNames(row); 
193 		for(int i=0;i<k.length;i++)
194 		{
195 			d.put(k[i], new String[] {String.valueOf(row.get(k[i]))});
196 		}
197 		this.addData(d);
198 	}

200 	/**
201 	 * Adds all the parameters passed to the method to the internal list. Checks for existing immutability. Resets {@link #keys} and {@link #immutableKeys} indices.
202 	 * @param paramsToAdd the parameters to add to the entry
203 	 */
204 	public void addAllParams(List<YADAParam> paramsToAdd) {
205 		for(YADAParam param : paramsToAdd)
206 		{
207 			if(!hasNonOverrideableParam(param.getName()))
208 			{
209 				addParam(param);
210 			}
211 		}
212 	}
213 	
214 	/**
215 	 * Adds a parameter to the internal list.  Checks for existing immutability. Resets {@link #keys} and {@link #immutableKeys} indices.
216 	 * @param param the parameter to add to the entry
217 	 */
218 	public void addParam(YADAParam param) {
219 		this.getParams().add(param);
220 		if(!hasNonOverrideableParam(param.getName()))
221 		{
222 			this.getParams().add(param);
223 			setKeys();
224 		}
225 	}

268 	private boolean hasParam(String key) {
269 		return this.keys.containsKey(key);
270 	}

272 	/**
273 	 * @param key the name of the parameter to check for immutabilitiy
274 	 * @return {@code true} if there is a param with the name equal to {@code key} and a {@code rule} equal {@link YADAParam#NONOVERRIDEABLE}
275 	 */
276 	private boolean hasNonOverrideableParam(String key) {
277 		return this.immutableKeys.containsKey(key);
