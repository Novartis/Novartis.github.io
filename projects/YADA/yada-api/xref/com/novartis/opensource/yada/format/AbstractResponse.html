194 	/**
195 	 * This method returns the appropriate converter {@link Converter} by looking first at the {@link YADARequest#PS_CONVERTER}
196 	 * parmater, if no class is set, or the class provided cannot be instantiated, the default converter is set 
197 	 * using {@link #getDefaultConverter(YADAQueryResult)}
198 	 * @param yqr the result container
199 	 * @return the appropriate Converter object
200 	 * @throws YADARequestException when the default converter cannot be instantiated
201 	 * @throws YADAConverterException when result reformatting fails
202 	 */
203 	protected Converter getConverter(YADAQueryResult yqr) throws YADARequestException, YADAConverterException
204 	{
205 		String converterClass = yqr.getYADAQueryParamValue(YADARequest.PS_CONVERTER);
206 		
207 		Converter converter;
208 		
209 		if( converterClass != null && !"".equals(converterClass))
210 		{
211 		  Constructor<?> c;
212 		  Class<?> clazz;
213 			try
214 			{
215 			  clazz = Class.forName(converterClass);
216 			  c = clazz.getConstructor(YADAQueryResult.class);
217 			  c.setAccessible(true);
218 			  converter = (Converter) c.newInstance(yqr);
219 			} 
220 			catch (Exception e)
221 			{
222 				l.warn("The specified class ["+converterClass+"] could not be instantiated.  Trying FQCN."); 
223 				try
224 				{
225 				  converterClass = FORMAT_PKG+converterClass;
226 				  clazz = Class.forName(converterClass);
227 				  c = clazz.getConstructor(YADAQueryResult.class);
228 	        c.setAccessible(true);
229 	        converter = (Converter) c.newInstance(yqr);
230 				}
231 				catch(Exception e1)
232 				{
233 					l.warn("The specified class ["+converterClass+"] could not be instantiated.  Trying default classes.",e1);
234 					converter = getDefaultConverter(yqr);
235 				}
236 			} 
237 		}
238 		else
239 		{
240 			converter = getDefaultConverter(yqr);
241 		}
242 		return converter;
243 	}

245 	/**
246 	 * Derives the default converter using the format and protocol values.
247 	 * @param yqr the result container
248 	 * @return the default Converter
249 	 * @throws YADARequestException when the default converter cannot be instantiated
250 	 * @throws YADAConverterException when result reformatting fails
251 	 */
252 	protected Converter getDefaultConverter(YADAQueryResult yqr) throws YADARequestException, YADAConverterException
253 	{
254 		Converter converter = null;
255 		String format    = yqr.getYADAQueryParamValue(YADARequest.PS_FORMAT);
256 		String type      = "";
257 		String className = "";
258 		try
259 		{
260 		  if(yqr.isFormatStructured())
261 	    {
262 		    format = format.toUpperCase();
263 	    }
264 		  else
265       {
266         format = DELIMITED;
267       }
268 		  
269 			if(getProtocol().equals(Parser.JDBC))
270 			{
271 			  type = RESULTSET_RESULT;
272 			}
273 			else if(getProtocol().equals(Parser.SOAP))
274 			{
275 			  type = SOAP_RESULT;
276 			}
277 			else if(getProtocol().equals(Parser.REST))
278 			{
279 			  type = REST_RESULT;
280 			}
281 			else if(getProtocol().equals(Parser.FILE))
282 			{
283 			  type = FILESYSTEM_RESULT;
284 			}
285 			else
286 			{
287 				String msg = "The converter you are attempting to instantiate requires a protocol or class that is not supported.  This could be a configuration issue.";
288 				throw new YADAConverterException(msg);
289 			}
290 			className = FORMAT_PKG+type+format+CONVERTER; 
291 	    Constructor<?> c = Class.forName(className).getConstructor(new Class[] { YADAQueryResult.class });
292 	    c.setAccessible(true);
293 	    converter = (AbstractConverter) c.newInstance(new Object[] {yqr});
294 		}
295 		catch (InstantiationException e)
296 		{
297 			String msg = "Could not instantiate Converter.";
298 			throw new YADARequestException(msg,e);
299 		} 
300 		catch (IllegalAccessException e)
301 		{
302 			String msg = "Could not access Converter class.";
303 			throw new YADARequestException(msg,e);
304 		} 
305 		catch (ClassNotFoundException e)
306 		{
307 			String msg = "Could not find Converter class";
308 			throw new YADARequestException(msg,e);
309 		} 
310 		catch (NoSuchMethodException e) 
311 		{
312 		  String msg = "There appears to be no Converter constructor accepting a YADAQueryResult parameter.";
313 		  throw new YADARequestException(msg,e);
314 		} 
315 		catch (SecurityException e) 
316 		{
317 		  String msg = "There appears to be no public Converter constructor accepting a YADAQueryResult parameter.";
318       throw new YADARequestException(msg,e);
319     } 
320 		catch (IllegalArgumentException e) 
321 		{
322 		  String msg = "There appears to be no public Converter constructor accepting a YADAQueryResult parameter.";
323       throw new YADARequestException(msg,e);
324     } 
325 		catch (InvocationTargetException e) 
326 		{
327 		  String msg = "There appears to be no public Converter constructor accepting a YADAQueryResult parameter.";
328       throw new YADARequestException(msg,e);
329     }		
330 		return converter;
331 	}

333 	/**
334 	 * The {@link #yadaQueryResults} array serves as a proxy for the queries being handled in the response.
335 	 * Since each {@link YADAQuery} object maps to a single {@link YADAQueryResult} this works out fine.
336 	 * Thus the method returns {@code true} if the {@link #yadaQueryResults} array has a {@code length} = 1.
337 	 * @return {@code true} if the {@link #yadaQueryResults} array has a {@code length} = 1
338 	 */
339 	protected boolean hasSingularQuery() {
340 		if(getYadaQueryResults().length == 1)
341 			return true;
342 		return false;
343 	}
344 		
345 	/**
346 	 * This method interrogates the {@link #yadaQueryResults} array. If {@link #hasSingularQuery()} returns {@code true} 
347 	 * then the value at index 1 is checked.  If this value is {@code null} or {@link YADAQueryResult#hasSingularResult()}
348 	 * returns {@code true} then this method returns {@code true} as well.
349 	 *  
350 	 * @return {@code true} if the {@link #yadaQueryResults} array contains a null value at it's first index or the result 
351 	 * object contains only a single result set 
352 	 */
353 	protected boolean hasSingularResult() {
354 		YADAQueryResult[] yqrs = getYadaQueryResults();
355 		if(hasSingularQuery())
356 		{  
357 			if(yqrs[0] == null
358 		   	||(yqrs[0] != null && yqrs[0].hasSingularResult()))
359 			{		
360 				return true;
361 			}
362 		}
363 		return false;
364 	}
365 	
366 	/**
367 	 * Returns {@code true} if {@link #hasSingularResult()} returns {@code false} and {@link #getHarmonyMap()} returns {@code null}
368 	 * @return {@code true} if {@link #hasSingularResult()} returns {@code false} and {@link #getHarmonyMap()} returns {@code null}
369 	 */
370 	protected boolean hasMultipleResults() {
371 		return !hasSingularResult() && getHarmonyMap() == null;
