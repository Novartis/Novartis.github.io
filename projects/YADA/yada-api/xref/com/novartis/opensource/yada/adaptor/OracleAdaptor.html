28  /**
29   * JDBCAdaptor class for execution of Oracle SQL queries.  This is a subclass of {@link JDBCAdaptor} but only 
30   * overrides the constructors.  {@link JDBCAdaptor}, thought abstract, was initially created as an
31   * Oracle adaptor, and by and large, contains generic code.
32   * 
33   * @author David Varon
34   *
35   */
36  public class OracleAdaptor extends JDBCAdaptor {

58  	/**
59  	 * Create a YADA SELECT statement from {@code core} wrapping it in two subqueries to account for pagination, sorting, and 
60  	 * filtering with a where clause. 
61  	 * @param core the code to wrap
62  	 * @param sortKey the column on which to sort
63  	 * @param sortOrder the sort order 'asc' or 'desc'
64  	 * @param firstRow the lower limiter of the result set
65  	 * @param pageSize the number of rows to return
66  	 * @param filters a JSON object containing the WHERE criteria 
67  	 * @return StringBuffer containing newly wrapped core sql, with pagination, filtering, and sorting, as desired
68  	 * @throws YADAAdaptorException when the query filters can't be converted into a WHERE clause
69  	 */
70  	@Override
71  	public StringBuffer buildSelect(String core, String sortKey, String sortOrder, int firstRow, int pageSize, JSONObject filters) throws YADAAdaptorException 
72  	{
73  		StringBuffer sql = new StringBuffer(SQL_SELECT_ALL);
74  		sql.append(SQL_FROM);
75  		sql.append(OPEN_PAREN);
76  		sql.append(NEWLINE+SPACE+SPACE);
77  		sql.append(SQL_SELECT);
78  		sql.append(SQL_CORE_ALIAS+"."+SQL_ALL);
79  		sql.append(COMMA+SPACE);
80  		sql.append(ROWNUM+SPACE+ROWNUM_ALIAS); /*  THIS IS AN ORACLE SPECIFIC LINE */
81  		sql.append(NEWLINE+SPACE+SPACE);
82  		sql.append(SQL_FROM);
83  		sql.append(SPACE+OPEN_PAREN+NEWLINE);
84  		sql.append(core);
85  		if (null != sortKey && !sortKey.equals(EMPTY))
86  		{
87  			if (core.toUpperCase().indexOf(SQL_ORDER_BY) == -1)
88  			{
89  				sql.append(NEWLINE);
90  				sql.append(SQL_ORDER_BY);
91  			}
92  			else
93  			{
94  				sql.append(COMMA+SPACE);
95  			}
96  			sql.append(sortKey);
97  			if (null != sortOrder && !sortOrder.equals(EMPTY))
98  			{
99  				 sql.append(SPACE + sortOrder);
100 			}
101 		}
102 		sql.append(NEWLINE+"       "+CLOSE_PAREN+SPACE);
103 		sql.append(SQL_CORE_ALIAS);
104 		sql.append(NEWLINE+SPACE+SPACE);
105 		sql.append(SQL_WHERE);
106 		sql.append(ROWNUM+SPACE+"< " + String.valueOf(firstRow+pageSize)); /*  THIS IS AN ORACLE SPECIFIC LINE */
107 		if (filters != null)
108 		{
109 			sql.append(NEWLINE+"    ");
110 			sql.append(SQL_AND);
111 			sql.append(NEWLINE+"    "+OPEN_PAREN+NEWLINE);
112 			sql.append(getQueryFilters(false));
113 			sql.append(NEWLINE+CLOSE_PAREN+NEWLINE);
114 		}
115 		sql.append(CLOSE_PAREN+SPACE+SQL_WRAPPER_ALIAS);
116 		sql.append(NEWLINE);
117 		sql.append(SQL_WHERE);
118 		sql.append(ROWNUM_ALIAS+SPACE+">="+SPACE+firstRow); /*  THIS IS AN ORACLE SPECIFIC LINE */
119 		return sql;
120 	}

122 	/**
123 	 * Returns a StringBuffer with core sql + filters, wrapped in an outer sql COUNT(*) query.  
124 	 * This is typically used in pagination scenarios to return the total number of records 
125 	 * returnable by a query when only a subset are requested by the application.
126 	 * 
127 	 * @param core the SQL to wrap in an outer count(*) query
128 	 * @param filters a JSON object containing the WHERE criteria
129 	 * @return StringBuffer of wrapped core sql including filters
130 	 * @throws YADAAdaptorException when the query filters can't be converted into a WHERE clause
131 	 */
132 	@Override
133 	public StringBuffer buildSelectCount(String core, JSONObject filters) throws YADAAdaptorException
134 	{
135 		boolean hasFilter = false;
136 		StringBuffer sql = new StringBuffer(SQL_SELECT);
137 		sql.append(SQL_COUNT_ALL);
138 		sql.append(SQL_COUNT);
139 		sql.append(NEWLINE);
140 		sql.append(SQL_FROM);
141 		sql.append(SPACE+SPACE+OPEN_PAREN+NEWLINE);
142 		sql.append(core);
143 		sql.append(NEWLINE+"       "+CLOSE_PAREN+SPACE);
144 		sql.append(SQL_CORE_ALIAS);
145 		if (filters != null)
146 		{
147 			sql.append(NEWLINE+SPACE+SPACE);
148 			sql.append(SQL_WHERE);
149 			sql.append(getQueryFilters(false));
150 			hasFilter = true;
151 		}
152 		if (this.yadaReq.getViewLimit() > -1)
153 		{
154 			if (hasFilter) 
155 			{ 
156 				sql.append(NEWLINE+SPACE+SPACE);
157 				sql.append(SQL_AND); 
158 			} 
159 			else 
160 			{ 
161 				sql.append(NEWLINE);
162 				sql.append(SQL_WHERE); 
163 			}
164 			
165 			sql.append("ROWNUM <=" + this.yadaReq.getViewLimit()); /*  THIS IS AN ORACLE SPECIFIC LINE */
166 		}
167 		return sql;
168 	}

170 	/**
171    * Enables checking for {@link JDBCAdaptor#ORACLE_TIMESTAMP_FMT} if {@code val} does not conform to {@link JDBCAdaptor#STANDARD_TIMESTAMP_FMT}
172    * @since 0.5.1.1
173    */
174   @Override
175   protected void setTimestampParameter(PreparedStatement pstmt, int index, char type, String val) throws SQLException 
176   {
177     if (EMPTY.equals(val) || val == null)
178     {
179       pstmt.setNull(index, java.sql.Types.DATE);
180     }
181     else
182     {
183       SimpleDateFormat sdf     = new SimpleDateFormat(STANDARD_TIMESTAMP_FMT);
184       ParsePosition    pp      = new ParsePosition(0);
185       Date             dateVal = sdf.parse(val,pp);
186       if(dateVal == null)
187       {
188         sdf     = new SimpleDateFormat(ORACLE_TIMESTAMP_FMT);
189         pp      = new ParsePosition(0);
190         dateVal = sdf.parse(val,pp);
191       }
192       if (dateVal != null)
193       {
194         long t = dateVal.getTime();
195         java.sql.Timestamp sqlDateVal = new java.sql.Timestamp(t);
196         pstmt.setTimestamp(index, sqlDateVal);
197       }
198     }
199   }
