97  	/**
98  	 * Currently supports directory READ, only that return single-level depth
99  	 * and WRITE and APPEND operations for single files. WRITEs can be performed
100 	 * on non-existent files, and currently there is no security attached to 
101 	 * ensure non-executables only can be written.
102 	 * @throws YADAAdaptorExecutionException when the execution fails as a result of an IO error or Resource error, e.g., a problem accessing or reading a file
103 	 */
104 	//TODO support parameters for filesystem depth
105 	@Override
106 	public void execute(YADAQuery yq) throws YADAAdaptorExecutionException
107 	{
108 		Object result = null;
109 		resetCountParameter(yq);
110 		for(int row=0;row<yq.getData().size();row++)
111 		{
112 			yq.setResult();
113 			YADAQueryResult yqr    = yq.getResult();
114 			String          urlStr = yq.getUrl(row);
115 			
116 			for(int i=0;i<yq.getParamCount(row);i++)
117 			{
118 				String param = "";
119 				Matcher m = PARAM_URL_RX.matcher(urlStr);
120 				if(m.matches())
121 				{
122 					param = yq.getVals(row).get(i);
123 					if(i==yq.getParamCount(row)-1                                 // last param
124 							&& (yq.getType().equals(QueryUtils.WRITE) || yq.getType().equals(QueryUtils.APPEND))) // non-read
125 					{
126 						urlStr = urlStr.replaceFirst(NON_READ_SFX_RX, "");
127 						setData(param);
128 					}
129 					else
130 					{
131 						urlStr = urlStr.replaceFirst(PARAM_SYMBOL_RX,m.group(1)+param);
132 					}
133 				}
134 				
135 			}
136 			
137 			File f = new File(urlStr.replace(PROTOCOL, ""));
138 			
139 			//TODO Prevent writing of unix executables. does this require simply checking for hashbang?
140 			//TODO Ensure no written files have x privs
141 			//TODO Enable creation of directories, maybe.
142 			if(yq.getType().equals(QueryUtils.WRITE) || yq.getType().equals(QueryUtils.APPEND))
143 			{
144 				boolean append = yq.getType().equals(QueryUtils.APPEND);
145 				try(FileWriter out = new FileWriter(f,append)) 
146 				{
147 			    
148 			    out.write(getData());
149 			    out.flush();
150 			  } 
151 				catch (IOException e) 
152 				{
153 					String msg = "There was a problem writing the file.";
154 					throw new YADAAdaptorExecutionException(msg,e);
155 				}
156 			}
157 			else
158 			{
159 			  if(f.isDirectory())
160 			  {
161 			    result = FileUtils.getFileList(f,0);
162 			  }
163 			  else
164 			  {
165 			    try 
166 			    {
167             result = FileUtils.getText(f);
168           } 
169 			    catch (YADAIOException e) 
170 			    {
171 			      throw new YADAAdaptorExecutionException(e);
172           }
173 			    catch (YADAResourceException e) 
174           {
175             throw new YADAAdaptorExecutionException(e);
176           }
177 			  }
178 			}
179 			yqr.addResult(row, result);
180 		}		
181 	}
