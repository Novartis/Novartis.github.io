209 	/**
210 	 * Retrieves the adaptor class from the application context given the
211 	 * parameter values.
212 	 * 
213 	 * @param source
214 	 *          the JNDI string or url mapped to the query's app in the YADA index
215 	 * @param version
216 	 *          the version of the framework, for selection of the proper adaptor
217 	 * @return the {@link Class} of the appropriate adaptor
218 	 * @throws YADAResourceException
219 	 *           when {@code source} can't be found, or there is an issue with the
220 	 *           application context
221 	 * @throws YADAUnsupportedAdaptorException
222 	 *           when the adaptor class mapped to {@code source} can't be found
223 	 */
224 	@SuppressWarnings("unchecked")
225   public Class<Adaptor> getAdaptorClass(String source, String version) throws YADAResourceException, YADAUnsupportedAdaptorException
226 	{
227 		String driverName = "";
228 		String className = REST_ADAPTOR_CLASS_NAME;
229 		l.debug("JNDI source is [" + source + "]");
230 		if (source.matches(RX_JDBC))
231 		{
232 			Context ctx;
233 			try
234 			{
235 				ctx = new InitialContext();
236 			} 
237 			catch (NamingException e)
238 			{
239 				String msg = "Could not create context.";
240 				throw new YADAResourceException(msg, e);
241 			}
242 			DataSource ds;
243 			try
244 			{
245 				ds = (DataSource)ctx.lookup(source);
246 			} 
247 			catch (NamingException e)
248 			{
249 				String msg = "Could not find data source at " + source;
250 				throw new YADAResourceException(msg, e);
251 			}
252 			//TODO checkout DriverManager as a preferred alternative to BasicDataSource for acquiring driverName.
253 			//TODO add integration tests for multiple containers after breaking tomcat dbcp dependency
254 			//     http://docs.oracle.com/javase/1.5.0/docs/api/java/sql/DriverManager.html?is-external=true
255 			driverName = ((BasicDataSource)ds).getDriverClassName();
256 			l.debug("JDBC driver is [" + driverName + "]");
257 			className = Finder.getEnv("adaptor/" + driverName + version);
258 		} 
259 		else if (source.matches(RX_SOAP))
260 		{
261 			className = SOAP_ADAPTOR_CLASS_NAME;
262 		} 
263 		else if (source.matches(RX_FILE))
264 		{
265 			className = FILESYSTEM_ADAPTOR_CLASS_NAME;
266 		}
267 		l.debug("JDBCAdaptor class is [" + className + "]");
268 
269 		Class<Adaptor> adaptorClass;
270 		try
271 		{
272 			adaptorClass = (Class<Adaptor>)Class.forName(className);
273 		} 
274 		catch (ClassNotFoundException e)
275 		{
276 			String msg = "Could not find appropriate adaptor class";
277 			throw new YADAUnsupportedAdaptorException(msg, e);
278 		} 
279 		catch (NoClassDefFoundError e)
280 		{
281 			String msg = "Could not find appropriate adaptor class";
282 			throw new YADAUnsupportedAdaptorException(msg, e);
283 		}
284 		return adaptorClass;
285 	}

334 			throw new YADAUnsupportedAdaptorException(msg, e);
335 		} catch (IllegalAccessException e)
336 		{
337 			String msg = "Error instanting adaptor for class" + adaptorClass.getName();
338 			throw new YADAUnsupportedAdaptorException(msg, e);
339 		} 
340 		catch (IllegalArgumentException e)
341 		{
342 			String msg = "Error instanting adaptor for class" + adaptorClass.getName();

344 		} 
345 		catch (SecurityException e)
346 		{
347 			String msg = "Error instanting adaptor for class" + adaptorClass.getName();
348 			throw new YADAUnsupportedAdaptorException(msg, e);
349 		}

351 		{
352 			String msg = "Error instanting adaptor for class" + adaptorClass.getName();

356 		{

453 	 * Utility wrapper method to manage {@link Parser} instantiation and parsing.
454 	 * 
455 	 * @param yq
456 	 *          The query object containing the SQL to parse
457 	 * @throws YADAParserException
458 	 *           when the parser fails
459 	 * @see com.novartis.opensource.yada.QueryManager
460 	 */
461 	@SuppressWarnings("static-method")
462 	private void processJDBCStatement(YADAQuery yq) throws YADAParserException
463 	{
464 		Parser parser = new Parser();
465 		Hashtable<String,String[]> ht = null;
466 
467 		ht = parser.parse(yq.getConformedCode());
468 		if (ht.size() > 0)
469 		{
470 			yq.setType(ht.get(Parser.TYPE)[0]);
471 			yq.setColumns(ht.get(Parser.COLUMNS));
472 			yq.setIns(ht.get(Parser.IN_COLUMNS));
473 			yq.setParameterizedColumns(ht.get(Parser.JDBC_COLUMNS));
474 		}
475 
476 		// return ht;
477 	}

479 	/**
480 	 * Called when the {@link CCJSqlParserManager#parse(java.io.Reader)} throws a
481 	 * {@link JSQLParserException} Uses regular expression to infer the query
482 	 * type.
483 	 * 
484 	 * @param yq
485 	 *          the query object containing the code to parse
486 	 * @throws YADAUnsupportedAdaptorException
487 	 *           when the adaptor can't be found or instantiated
488 	 */
489 
490 	// private Hashtable<String,String[]> processUnparsableStatement(String code)
491 	public void processStatement(YADAQuery yq) throws YADAUnsupportedAdaptorException
492 	{
493 		String code = getConformedCode(yq.getYADACode());
494 		Class<Adaptor> adaptorClass = yq.getAdaptorClass();
495 		if (isJdbc(yq.getAdaptorClass()))
496 		{
497 			try
498 			{
499 				processJDBCStatement(yq);
500 			} 
501 			catch (YADAParserException e)
502 			{
503 				l.warn("Attempting to qualify previously unparsable statement");
504 				if (isCallable(code))
505 					yq.setType(Parser.CALL);
506 				else if (isSelect(code))
507 					yq.setType(Parser.SELECT);
508 				else if (isUpdate(code))
509 					yq.setType(Parser.UPDATE);
510 				else if (isInsert(code))
511 					yq.setType(Parser.INSERT);
512 				else if (isDelete(code))
513 					yq.setType(Parser.DELETE);
514 			}
515 		} 
516 		else if (isSoap(adaptorClass))
517 		{
518 			yq.setType(Parser.SOAP);
519 		} 
520 		else if (isRest(adaptorClass))
521 		{
522 			yq.setType(Parser.REST);
523 		} 
524 		else if (isFileSystem(adaptorClass))
525 		{
526 			if (isRead(code))
527 				yq.setType(READ);
528 			else if (isWrite(code))
529 				yq.setType(WRITE);
530 			else if (isAppend(code))
531 				yq.setType(APPEND);
532 		} 
533 		else
534 		{
535 			String msg = "The query you are attempting to execute requires a protocol or class that is not supported.  This could be a configuration issue.";
536 			throw new YADAUnsupportedAdaptorException(msg);
537 		}
538 	}

540 	/**
541 	 * Interrogates {@code yq} for the adaptor class and sets the protocol
542 	 * attribute accordingly.
543 	 * 
544 	 * @param yq
545 	 *          the query object in which to set the protocol attribute
546 	 * @throws YADAUnsupportedAdaptorException
547 	 *           when the adaptor class cannot be found
548 	 */
549 	public void setProtocol(YADAQuery yq) throws YADAUnsupportedAdaptorException
550 	{
551 		Class<Adaptor> adaptorClass = yq.getAdaptorClass();
552 		if (isJdbc(adaptorClass))
553 		{
554 			yq.setProtocol(Parser.JDBC);
555 		}
556 		else if (isSoap(adaptorClass))
557 		{
558 			yq.setProtocol(Parser.SOAP);
559 		} 
560 		else if (isRest(adaptorClass))
561 		{
562 			yq.setProtocol(Parser.REST);
563 		} 
564 		else if (isFileSystem(adaptorClass))
565 		{
566 			yq.setProtocol(Parser.FILE);
567 		}
568 		else
569 		{
570 			String msg = "The query you are attempting to execute requires a protocol or class that is not supported.  This could be a configuration issue.";
571 			throw new YADAUnsupportedAdaptorException(msg);
572 		}
573 		yq.addParam(YADARequest.PS_PROTOCOL, yq.getProtocol());
574 	}

607 	/**
608 	 * Returns {@code true} if the query content matches an SQL UPDATE statement
609 	 * syntax (see {@link #RX_UPDATE}.
610 	 * 
611 	 * @param code
612 	 *          stored code (with YADA markup)
613 	 * @return {@code true} if the query content matches an SQL UPDATe statement
614 	 *         syntax
615 	 */
616 	public boolean isUpdate(String code)
