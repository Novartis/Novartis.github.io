148 	/**
149 	 * Retreives the environment variables mapped to the application context using JNDI, for example, {@code "io/in"}, or {@code "yada_bin"}.  
150 	 * If {@code jndiPath} is not found in the application context, an ottempt is made to find a system property with the same name. 
151 	 * @param jndiPath a string registered in the application context, or system property
152 	 * @return {@link String} value of variable mapped to JNDI parameter string
153 	 * @throws YADAResourceException when the JNDI string is not found in the application context or system properties
154 	 */
155 	public static String getEnv(String jndiPath) throws YADAResourceException
156 	{
157 		InitialContext ictx;
158 		String result = "";
159 		try 
160 		{
161 			ictx = new javax.naming.InitialContext();
162 			javax.naming.Context env = (javax.naming.Context)ictx.lookup(JNDI_PREFIX);
163 			try
164 			{
165 				NamingEnumeration<NameClassPair> list = ictx.list(JNDI_PREFIX);
166 				l.debug("JNDI Environment:");
167 				while (list.hasMore())
168 				{
169 					NameClassPair nc = list.next();
170 					l.debug("    ["+nc+"]");
171 				}
172 			}
173 			catch (Exception e)
174 			{
175 				l.debug("An exception was thrown while attempting to list contexts. The context implementation might not support it.  This is a non-fatal error");
176 			}
177 			result = (String) env.lookup(jndiPath);
178 		} 
179 		catch (NamingException e) 
180 		{
181 			
182 			String msg = "There was a problem locating the resource or variable identified by the supplied JNDI path ("+jndiPath+") in the initial context...trying system properties.";
183 			l.warn(msg);
184 			result = System.getProperty(jndiPath);
185 			if(result == null)
186 			{
187 				msg = "The property [" + jndiPath + "] was not found.";
188 				throw new YADAResourceException(msg,e);
189 			}
190 		} 
191 		return result;
192 	}

203 	/**
204 	 * Retrieves the query {@code q} from the distributed index (i.e., database).  Typically the query is 
205 	 * then added to the in-memory cache. 
206 	 * @param q the query name to retrieve
207 	 * @return the {@link YADAQuery} retrieved from the distributed index
208 	 * @throws YADAFinderException if the query {@code q} can't be found in the YADA index.  Check your spelling and case.
209 	 * @throws YADAConnectionException if an error occurs when attempting to query the YADA Index.  This could be the result of a configuration issue.
210 	 * @see #getQuery(String)
211 	 */
212 	public YADAQuery getQueryFromIndex(String q) throws YADAConnectionException, YADAFinderException
213 	{
214 		ResultSet rs = null;
215 		YADAQuery yq = new YADAQuery();
216 		try
217 		{
218 			
219 			Connection        conn  = ConnectionFactory.getConnection(getYADAJndi());
220 			PreparedStatement pstmt;
221 			try
222 			{
223 				pstmt = conn.prepareStatement(YADA_PKG_SQL);
224 				pstmt.setString(1,q);
225 			} 
226 			catch (SQLException e)
227 			{
228 				String msg = "Unable to create or configure the PreparedStatement used to lookup the requested query in the YADA Index.  This could be a serious configuration issue.";
229 				throw new YADAConnectionException(msg,e);
230 			}
231 			int row = 0;
232 			
233 			try
234 			{
235 				rs = pstmt.executeQuery();
236 				if(!rs.isBeforeFirst())
237 				{
238 					String msg = "The request query ["+q+"] does not exist.";
239 					throw new YADAFinderException(msg);
240 				}
241 				while (rs.next())
242 				{
243 					if (row == 0)
244 					{
245 						String v = rs.getString(YADA_VERSION);
246 						yq.setVersion(v == null ||  v.equals(NOT_APPLICABLE) ? "" : v);
247 						yq.setCoreCode(rs.getString(YADA_QUERY));
248 						yq.setSource(rs.getString(YADA_SOURCE));
249 						yq.setQname(q);
250 					}
251 					setDefaultParam(yq,rs);
252 					row++;
253 				}
254 			} 
255 			catch (SQLException e)
256 			{
257 				String msg = "The lookup query caused an error. This could be because the query name ("+q+") was mistyped or doesn't exist in the YADA Index";
258 				throw new YADAFinderException(msg,e);
259 			}
260 			l.debug("Query package: "+ yq.toString());
261 		}
262 		finally
263 		{
264 			ConnectionFactory.releaseResources(rs);
265 		}
266 		return yq;
267 	}

281 	/**
282 	 * A cornerstone to the entire YADA framework, this method takes the name of a stored 
283 	 * query as an argument and returns a {@link YADAQuery} object containing the query code and default parameters,
284 	 * as well as data structures and methods to facilitate it's execution.
285 	 * <p>
286 	 * There is now an <a href="http://www.ehcache.org">EhCache</a> implementation in which all queries are stored upon retrieval.
287 	 * When a query is subsequently requested, it is found in the cache, cloned, and the clone put to use.
288 	 * </p>
289 	 *
290 	 * 
291 	 * @since 0.4.0.0
292 	 * @param q the stored canonical name of the desired {@link YADAQuery} 
293 	 * @param updateStats set to {@code true} to execute parallel operation to query update access count and date, {@code false} to suppress the operation
294 	 * @return {@link YADAQuery} object encapsulating the query code and parameters
295 	 * @throws YADAFinderException if the query {@code q} can't be found in the YADA index.  Check your spelling and case.
296 	 * @throws YADAConnectionException if an error occurs when attempting to query the YADA Index.  This could be the result of a configuration issue. 
297 	 * @see YADAQuery 
298 	 *  
299 	 */
300 	public YADAQuery getQuery(String q, boolean updateStats) throws YADAConnectionException, YADAFinderException
301 	{
302 		final String qname = q;
303 		YADAQuery yq        = null; 
304 		Element   cachedYq  = null;
305 		Cache     yadaIndex = ConnectionFactory.getCacheConnection(YADA_CACHE_MGR,YADA_CACHE);
306 		
307 		if(yadaIndex != null)
308 		{  
309 		  cachedYq = yadaIndex.get(qname);
310   		if(cachedYq != null)
311   		{
312   			yq = new YADAQuery((YADAQuery)cachedYq.getObjectValue());
313   			l.debug("YADAQuery ["+qname+"] retrieved from cache");
314   		}
315   		else
316   		{
317   			yq = getQueryFromIndex(q);
318   			// update cache
319   			Element yqElement = new Element(qname,yq);
320   			yadaIndex.put(yqElement);
321   			yq.setCached(true);
322   			l.debug("YADAQuery ["+qname+"] stored in cache.");
323   		}
324 		}
325 		
326 		if(updateStats)
327 		{
328 			// log usage of query 
329 			(new Thread() {
330 				
331 				@Override
332 				public void run() {
333 					try
334 					{
335 						updateQueryStatistics(qname);
336 					} 
337 					catch (YADAConnectionException e)
338 					{
339 						l.error(e.getMessage(),e);
340 					} 
341 					catch (YADAFinderException e)
342 					{
343 						l.error(e.getMessage(),e);
344 					} 
345 					finally
346 					{
347 						//TODO what happens in finally block?
348 					}
349 				}
350 			}).start();
351 		}
352 				
353 		return yq;
354 	}

356 	/**
357 	 * Wraps default parameter data stored in the YADAIndex into a {@link YADAParam} object and adds it to {@code yq}.
358 	 * @param yq the {@link YADAQuery} to which to add the parameter defined in {@code rs} 
359 	 * @param rs the data containing the parameter, as retrieved from the YADA Index
360 	 * @throws YADAFinderException when {@code rs} can't be processed
361 	 */
362 	@SuppressWarnings("static-method")
363 	private void setDefaultParam(YADAQuery yq, ResultSet rs) throws YADAFinderException
364 	{
365 		//TODO should this return null, rather than default 'na' values?
366 		try
367 		{
368 			String name = rs.getString(YADA_PARAMNAME);
369 			if(!( name == null || name.equals(NOT_APPLICABLE)))
370 			{	
371 				String target = rs.getString(YADA_PARAMTARGET) != null ? rs.getString(YADA_PARAMTARGET) : NOT_APPLICABLE;
372 				String value  = rs.getString(YADA_PARAMVAL) != null ? rs.getString(YADA_PARAMVAL) : NOT_APPLICABLE;
373 				YADAParam param = new YADAParam();
374 				param.setName(name);
375 				param.setTarget(target);
376 				param.setValue(value);
377 				param.setRule(rs.getInt(YADA_PARAMRULE));
378 				param.setDefault(true);
379 				yq.addParam(param);
380 			}
381 		}
382 		catch(SQLException e)
383 		{
384 			String msg = "Unable to set default params.";
385 			throw new YADAFinderException(msg,e);
386 		}
387 	}

389 	/**
390 	 * A utility method called in a separate thread by {@link Finder#getQuery(String)} to increment a query-access counter in the YADA Index.
391 	 * 
392 	 * @param qname the name of the query just requested
393 	 * @throws YADAConnectionException when the YADA Index can't be accessed
394 	 * @throws YADAFinderException when {@code qname} can't be found in the YADA Index.
395 	 * @see Finder#getQuery(String)
396 	 */
397 	void updateQueryStatistics(String qname) throws YADAConnectionException, YADAFinderException
398 	{
399 
400 		l.debug("Updating Query Stats for ["+qname+"]");
401 		PreparedStatement pstmt = null;
402 		String querySql    = SQL_STATS;
403 		try
404 		{
405 			Connection conn = ConnectionFactory.getConnection(JNDI_PREFIX + YADA_INDEX);
406 			
407 			try
408 			{
409 				pstmt = conn.prepareStatement(querySql);
410 				pstmt.setDate(1, new java.sql.Date(System.currentTimeMillis()));
411 				pstmt.setString(2,qname);
412 				
413 			} 
414 			catch (SQLException e)
415 			{
416 				String msg = "Unable to create or configure the PreparedStatement used to update the access statistics for the requested query in the YADA Index.  This could be a serious configuration issue.";
417 				throw new YADAConnectionException(msg,e);
418 			}
419 			
420 			try
421 			{
422 				pstmt.executeUpdate();
423 			}
424 			catch (SQLException e)
425 			{
426 				String msg = "The lookup query caused an error. This could be because the query name ("+qname+") was mistyped or doesn't exist in the YADA Index";
427 				throw new YADAFinderException(msg,e);
428 			}
429 			
430 		}
431 		finally
432 		{
433 			ConnectionFactory.releaseResources(pstmt);
434 		}
