17  import java.io.BufferedReader;
18  import java.io.IOException;
19  import java.io.InputStreamReader;
20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import org.apache.log4j.Logger;
24  import org.json.JSONArray;
25  import org.json.JSONException;
26  import org.json.JSONObject;
27  
28  import com.novartis.opensource.yada.Finder;
29  import com.novartis.opensource.yada.YADARequest;
30  import com.novartis.opensource.yada.YADARequestException;
31  import com.novartis.opensource.yada.YADAResourceException;

34  /**
35   * A java API enabling execution of pre-existing scripts, in pre-defined locations on the YADA server, as Preprocess plugins.
36   * For more information see <a href="../../../../../../../plugins.html">Plugin Guide</a>
37   * @author David Varon
38   *
39   */
40  public class ScriptPreprocessor extends AbstractPreprocessor {
41  	/**
42  	 * Local logger handle
43  	 */
44  	private static Logger l = Logger.getLogger(ScriptPreprocessor.class);
45  	
46  	/**
47  	 * Enables the execution of a script stored in the {@code YADA_BIN} directory.
48  	 * To execute a script preprocessor plugin, pass {@code preArgs}, or just {@code args}
49  	 * the first argument being the name of the script executable, and the rest of the arguments
50  	 * those, in order, to pass to it. If {@link YADARequest#getPreArgs()} is not null
51  	 * and {@link YADARequest#getPlugin()} is null, then the plugin will be set to 
52  	 * {@link YADARequest#SCRIPT_PREPROCESSOR} automatically.  
53  	 * <p>
54  	 * The script must return a JSON object with name/value pairs corresponding
55  	 * to {@link YADARequest} parameters. These name/value pairs are then marshaled into
56  	 * {@code yadaReq}, which is subsequently returned by the method.
57  	 * </p>
58  	 * @see com.novartis.opensource.yada.plugin.Bypass#engage(com.novartis.opensource.yada.YADARequest)
59  	 */
60  	@Override
61  	public YADARequest engage(YADARequest yadaReq) throws YADAPluginException 
62  	{
63  		List<String> cmds = new ArrayList<>();
64  		// get args
65  		List<String> args = yadaReq.getPreArgs().size() == 0 ? yadaReq.getArgs() : yadaReq.getPreArgs();
66  		// add plugin
67  		try 
68  		{
69  			cmds.add(Finder.getEnv("yada_bin")+args.remove(0));
70  		} 
71  		catch (YADAResourceException e)
72  		{
73  			String msg = "There was a problem locating the resource or variable identified by the supplied JNDI path (yada_bin) in the initial context.";
74  			throw new YADAPluginException(msg,e);
75  		}
76  		// add args
77  		cmds.addAll(args);
78  		for (String arg : args)
79  		{
80  			cmds.add(arg);
81  		}
82  		// add yadaReq json
83  		cmds.add(yadaReq.toString());
84  		l.debug("Executing script plugin: "+cmds);
85  		String scriptResult = "";
86  		String s            = null;
87  		try
88  		{
89  			ProcessBuilder builder = new ProcessBuilder(cmds);
90  			builder.redirectErrorStream(true);
91  			Process process = builder.start();
92  			try(BufferedReader si = new BufferedReader(new InputStreamReader(process.getInputStream())))
93  			{
94    			while ((s = si.readLine()) != null)
95    			{
96    				l.debug("  LINE: "+s);
97    				scriptResult += s;
98    			}
99  			}
100 			process.waitFor();
101 			JSONObject params = new JSONObject(scriptResult);
102 			for ( String param : JSONObject.getNames(params))
103 			{
104 				JSONArray ja     = (JSONArray) params.get(param);
105 				l.debug("JSON array "+ja.toString());
106 				// remove square brackets and leading/trailing quotes
107 				String    values = ja.toString().substring(2, ja.toString().length()-2);
108 				// remove "," between values and stuff in array
109 				String[]  value  = values.split("\",\"");
110 				l.debug("Value has " + value.length + " elements");
111 				yadaReq.invokeSetter(param, value);
112 			}
113 		}
114 		catch(IOException e)
115 		{
116 			String msg = "Failed to get input from InputStream.";
117 			throw new YADAPluginException(msg,e);
118 		}
119 		catch(InterruptedException e)
120 		{
121 			String msg = "The external process executing the script was interrupted.";
122 			throw new YADAPluginException(msg,e);
123 		}
124 		catch(JSONException e)
125 		{
126 		  String msg  = "Parameter configuration failed. The script return value should conform to the syntax:\n";
127 		         msg += "  {\"key\":[\"value\"...],...}.\n";
128 		         msg += "  The array syntax for values complies to the return value of HTTPServletResponse.getParameterMap().toString()";
129 			throw new YADAPluginException(msg,e);
130 		}
131 		catch (YADARequestException e)
132 		{
133 			String msg = "Unable to set parameters.";
134 			throw new YADAPluginException(msg,e);
135 		}
136 		
137 		return yadaReq;
138 	}
