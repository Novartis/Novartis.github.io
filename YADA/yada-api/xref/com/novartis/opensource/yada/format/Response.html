15  package com.novartis.opensource.yada.format;

17  import com.novartis.opensource.yada.YADAQueryResult;

19  /**
20   * The YADA API for creating formatted results.
21   * 
22   * @author David Varon
23   *
24   */
25  public interface Response {
26  	
27  	/**
28  	 * Use to access the raw results stored in {@code yadaQueryResults}, establish the Response structure
29  	 * and initiate population of the object via {@link #create()}, and fill the object via one of the {@code append} methods
30  	 * <p>
31  	 * The difference between {@link #compose(YADAQueryResult[])} and {@link #create()} is somewhat academic. {@code compose}
32  	 * enables the caller, typically {@link com.novartis.opensource.yada.Service} to pass the {@link YADAQueryResult}s associated to
33  	 * a {@link com.novartis.opensource.yada.YADAQuery} to the Response, whereas {@link #create()} does not.  Theoretically, {@link #compose(YADAQueryResult[])}
34  	 * could do everything {@link #create()} does, but the added granularity makes it a bit easier to code.
35  	 * </p>
36  	 * @param yadaQueryResults array containing result objects
37  	 * @return this Response object
38  	 * @throws YADAResponseException when response composition fails
39  	 * @throws YADAConverterException when result reformatting fails
40  	 * @see com.novartis.opensource.yada.format.JSONResponse
41  	 */
42  	public Response compose(YADAQueryResult[] yadaQueryResults) throws YADAResponseException, YADAConverterException;
43  	/**
44  	 * A method meant to be called by {@link #compose(YADAQueryResult[])} for filling the new Response object with 
45  	 * it's foundational attributes, i.e., root nodes, root keys, headers, etc.
46  	 * @return the Response object
47  	 * @throws YADAResponseException when response composition fails
48  	 * @throws YADAConverterException when result reformatting fails
49  	 */
50  	public Response create() throws YADAResponseException, YADAConverterException;
51  	/**
52  	 * Append the {@link String} {@code s} to the Response 
53  	 * @param s a {@link String} of content to append to the response
54  	 * @return the Response object
55  	 * @throws YADAResponseException when response composition fails
56  	 * @throws YADAConverterException when result reformatting fails
57  	 */
58  	public Response append(String s) throws YADAResponseException, YADAConverterException;
59  	/**
60  	 * Append the {@link java.lang.Integer} {@code i} to the Response 
61  	 * @param i a {@link java.lang.Integer} to append to the response
62  	 * @return the Response object
63  	 * @throws YADAResponseException when response composition fails
64  	 * @throws YADAConverterException when result reformatting fails
65  	 */
66  	public Response append(Integer i) throws YADAResponseException, YADAConverterException;
67  	/**
68  	 * Append the {@link java.lang.Object} {@code i} to the Response 
69  	 * @param o an {@link java.lang.Object} to append to the response
70  	 * @return the Response object
71  	 * @throws YADAResponseException when response composition fails
72  	 * @throws YADAConverterException when result reformatting fails
73  	 */
74  	public Response append(Object o) throws YADAResponseException, YADAConverterException;
75  	/**
76  	 * Get a string version of the Response object.  This should be callable by {@link com.novartis.opensource.yada.Service} to return
77  	 * the Response content to the client
78  	 * @param prettyPrint flag indicating whether or not to format a response string, for example, a json string, with indenting
79  	 * @return the Response as a {@link String}
80  	 * @throws YADAResponseException when response composition fails
81  	 */
82  	public String   toString(boolean prettyPrint) throws YADAResponseException;
83  	
84  }
