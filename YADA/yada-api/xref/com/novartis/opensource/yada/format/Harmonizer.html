39  /**
40   * Class containing utility methods for re-mapping JSON keys, result set columns, 
41   * flattening and pruning results, etc. The class hands off processing of these 
42   * transformations to Javascript via Rhino.
43   * @author Dave Varon
44   * @since 6.1.0
45   * @see "https://developer.mozilla.org/en-US/docs/Mozilla/Projects/Rhino"
46   */
47  public class Harmonizer 
48  {
49    private ScriptEngineManager factory;
50    @SuppressWarnings("unused")
51    private ScriptEngine engine;
52    private ScriptableObject global;
53    
54    /**
55     * A constant with value: {@value}
56     */
57    public static final String PRUNE = "prune";
58    
59    /**
60     * Default constructor which initializes javascript context and scope
61     * @throws YADAResourceException when the path to {@code r.js} or {@code harmony.js} is corrupt
62     * @throws YADAIOException when the path to {@code r.js} or {@code harmony.js} can't be read
63     * @throws YADAConverterException when the Rhino javascript engine throws an error
64     * @see "https://developer.mozilla.org/en-US/docs/Mozilla/Projects/Rhino/Scopes_and_Contexts"
65     */
66    public Harmonizer() throws YADAResourceException, YADAIOException, YADAConverterException 
67    {
68      this.factory       = new ScriptEngineManager();
69      this.engine        = this.factory.getEngineByName("JavaScript");
70      Context    ctx     = Context.enter();
71      this.global        = ctx.initStandardObjects(new JsRuntimeSupport(), true);
72      String[]   names   = { "print", "load" };
73      this.global.defineFunctionProperties(names, this.global.getClass(), ScriptableObject.DONTENUM);
74      Scriptable argsObj = ctx.newArray(this.global, new Object[] { });
75      this.global.defineProperty("arguments", argsObj, ScriptableObject.DONTENUM);
76      try(Reader inLo   = new InputStreamReader(getClass().getResourceAsStream("/utils/lodash.min.js"));
77          Reader inJSON = new InputStreamReader(getClass().getResourceAsStream("/utils/json2.js"));
78          Reader inHmap = new InputStreamReader(getClass().getResourceAsStream("/utils/harmony.js")))
79          //Reader inR    = new InputStreamReader(getClass().getResourceAsStream("/utils/r.js")))
80      {
81        //ctx.evaluateReader(this.global, inR , "require", 0, null);
82        ctx.evaluateReader(this.global, inLo, "_", 0, null);
83        ctx.evaluateReader(this.global, inJSON, "JSON", 0, null);
84        ctx.evaluateReader(this.global, inHmap, "harmony", 0, null);
85      } 
86      catch (MalformedURLException e) 
87      {
88        String msg = "One of the required resources could not be loaded from the provided path.";
89        throw new YADAResourceException(msg, e);
90      } 
91      catch (IOException e) 
92      {
93        String msg = "One of the required resources could not be read.";
94        throw new YADAIOException(msg, e);
95      }  
96      catch (EvaluatorException e)
97      {
98        String msg = "There was a problem with the Rhino Javascript engine.";
99        throw new YADAConverterException(msg, e);
100     }
101     finally
102     {
103       Context.exit();
104     }
105   }

107   /**
108    * Accepts a {@link String} name of the javascript function to execute, 
109    * which must be {@code harmonize} or {@code flatten}, and an array of 
110    * Objects (Strings) to pass to these functions.  For {@code harmonize}, 
111    * {@code o} must contain both the source string to transform, and the 
112    * harmony map. Both are expected to be JSON strings. The source string 
113    * can be a JSON array.  For {@code flatten} only the source is expected. 
114    * @param func the name of the javascript function to execute. Must be {@code harmonize} or {@code flatten}. 
115    * @param o the array of arguments to hand off to the js function
116    * @return the string result of the transformation.
117    * @throws YADAConverterException when the Rhino javascript engine throws an error
118    */
119   public String call(String func, Object[] o) throws YADAConverterException
120   {
121     Function f      = (Function)this.global.get(func,this.global);
122     String   result = "";
123     Context  ctx    = Context.enter();
124     try
125     {
126       result = f.call(ctx, this.global, this.global, o).toString();
127     } 
128     catch (EvaluatorException e)
129     {
130       String msg = "There was a problem with the Rhino Javascript engine.";
131       throw new YADAConverterException(msg, e);
132     }
133     finally
134     {
135       Context.exit();
136     }
137     
138     return result;
139   }
