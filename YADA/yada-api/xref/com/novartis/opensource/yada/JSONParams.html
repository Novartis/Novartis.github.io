25  /**
26   * A {@link LinkedHashMap} implementation for storing values passed to the YADA {@link Service}, by way of {@link YADARequest} 
27   * in the url parameters or http POST content.  Naturally, {@code JSONParams} or {@code j} parameters map directly, but in fact, standard parameters
28   * ({@code qname} or {@code q}, and {@code params} or {@code p}) are ultimately mapped to these objects as well. 
29   * 
30   * <p>This map contains YADA query names as keys, and {@link JSONParamsEntry} objects as values. Thus the structure is like the following:</p>
31   * <p><code>query : { JSONParamsEntry }</code></p>
32   * <p>which in turns parses out to:</p>
33   * <p><code>query : { DATA: [ array of data ], PARAMS: [ array of params ] }</code></p>
34   * 
35   * @since 0.4.0.0               
36   * @author David Varon
37   * @see JSONParamsEntry
38   */
39  public class JSONParams extends LinkedHashMap<String, JSONParamsEntry> {

55  	/**
56  	 * Takes a json string as an argument and converts it into a {@link JSONParams} object.
57  	 * The {@link String} must be a {@link String} reprentation of a {@link JSONArray}
58  	 * and must conform to the <a href="../../../../../../jsonparams.html">JSONParams specification</a>, e.g.:
59  	 * 
60  	 * <code>[{"qname":"name of query","DATA":[{"col_1":"val_1",..."col_n":"val_n"},{...},...]}</code> 
61  	 * 
62  	 * @since 0.4.0.0
63  	 * @param jp the json string to convert to an object
64  	 * @throws YADAQueryConfigurationException when the object can't be created, probably due to a malformed, or non-compliant json string.
65  	 */
66  	public JSONParams(String jp) throws YADAQueryConfigurationException 
67  	{
68  		try 
69  		{
70  			convertJSONObjects(new JSONArray(jp));
71  		} 
72  		catch (JSONException e) 
73  		{
74  			String msg = "Unable to create JSONParams object from supplied parameters.";
75  			throw new YADAQueryConfigurationException(msg, e);
76  		}
77  	}

79  	/**
80  	 * <p>Takes a {@link String}[] array as an argument and converts it into a {@link JSONParams} object.</p>
81  	 * 
82  	 * <p>The {@code jp} array can contain a representation of a {@link JSONArray}, in which case only the first index of the {@code jp} param will be processed.</p>
83  	 * 
84  	 * <p>Alternatively, if {@code jp} contains {@link String} representations of {@link JSONObject}s, 
85  	 * a new {@link JSONArray} will be instantiated and populated with the objects.</p>
86  	 * 
87  	 * <p>Currently, only JSON syntax checking is performed, but no YADA syntax validation against the <a href="">JSONParams specification</a> is performed.</p> 
88  	 * 
89  	 * This is the constructor called by {@link Service#handleRequest(javax.servlet.http.HttpServletRequest)} when encountering a JSONParams url parameter.
90  	 * @since 0.4.0.0
91  	 * @param jp an array containing json strings 
92  	 * @throws YADAQueryConfigurationException when {@code jp} is not convertable into a {@link JSONParams} object 
93  	 * @see YADARequest#setJsonParams(String[])
94  	 * @see Service#handleRequest(javax.servlet.http.HttpServletRequest)
95  	 */
96    
97  	public JSONParams(String[] jp) throws YADAQueryConfigurationException 
98  	{
99  		JSONObject jo;
100 		JSONArray  ja = new JSONArray();
101 		for(String j : jp)
102 		{
103 			try 	
104 			{
105 				ja = new JSONArray(j);
106 				break;
107 			} 
108 			catch (JSONException e) 
109 			{
110 				try
111 				{
112 					jo = new JSONObject(j);
113 					ja.put(jo);
114 				}
115 				catch (JSONException e1)
116 				{
117 					String msg = "Unable to create JSONParams object from supplied parameters.";
118 					throw new YADAQueryConfigurationException(msg, e1);
119 				}
120 			}
121 		}
122 		convertJSONObjects(ja);
123 	}

146 	/**
147 	 * Takes input that originated as a json array of json objects, builds {@link JSONParamsEntry} objects 
148 	 * for each of them, and adds them to itself, mapping to the appropriate query names
149 	 * @param ja an array of json objects conforming to the <a href="../../../../../../jsonparams.html">JSONParams specification</a>
150 	 * @throws YADAQueryConfigurationException when {@code ja} is not convertable into a {@link JSONParams} object
151 	 */
152 	private void convertJSONObjects(JSONArray ja) throws YADAQueryConfigurationException
153 	{
154 		try
155 		{
156 			for (int i = 0; i < ja.length(); i++) // multiple queries
157 			{
158 				// object with the query and data
159 				JSONObject      jobj  = ja.getJSONObject(i);
160 				String          qname = jobj.getString(YADARequest.getParamKeyVal("PL_QNAME"));
161 				if(qname == null)
162 				{
163 					qname = jobj.getString(YADARequest.getParamKeyVal("PS_QNAME"));
164 				}
165 				JSONParamsEntry entry = new JSONParamsEntry(jobj);
166 				put(qname,entry);
167 			}
168 			l.debug(this.toString());
169 		}
170 		catch (JSONException e) 
171 		{
172 			String msg = "Unable to create JSONParams object from supplied parameters.";
173 			throw new YADAQueryConfigurationException(msg, e);
174 		}
175 	}

274 	/**
275 	 * Convenience method for appending multiple rows to the internal {@link JSONParamsEntry} mapped to the {@code query}.
276 	 * If the {@code query} doesn't exist, it will be added via {@link JSONParams#addData(String, LinkedHashMap)}
277 	 * Uses {@link JSONParams#addRowForYADAQuery(String, JSONObject)} for convenience, to avoid dealing with value object casting
278 	 * @param query the name of the query to which to associate the datas
279 	 * @param fields list of field names to corresponding to {@code values}
280 	 * @param values list of values to associate to {@code fields}
281 	 * @throws JSONException if the internal JSONObject can not be created or populated.
282 	 */
283 	public void addRowsForYADAQuery(String query, List<String> fields, List<String[]> values) throws JSONException  {
284 		for (String[] vals : values)
285 		{
286 			JSONObject row = new JSONObject();
287 			for (int i=0; i<vals.length; i++)
288 			{
289 				row.put(fields.get(i),vals[i]);
290 			}
291 			addRowForYADAQuery(query, row);
292 		}
293 	}
