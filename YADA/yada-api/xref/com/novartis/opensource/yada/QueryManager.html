160 	 * A broker method which calls {@link #endowQuery(String)} (for standard param requests)
161 	 * or {@link #endowQueries(JSONParams)} for json params requests, followed by
162 	 * {@link #prepQueriesForExecution()} in succession
163 	 * 
164 	 * @throws YADAFinderException
165 	 *           when there is an issue retrieving a query from the YADA index
166 	 * @throws YADAConnectionException
167 	 *           when there is an issue opening a connection to a source
168 	 *           referenced by a query
169 	 * @throws YADAQueryConfigurationException
170 	 *           if request does not contain either a {@code qname} or {@code q},
171 	 *           or a {@code JSONParams} or {@code j} parameter
172 	 * @throws YADAUnsupportedAdaptorException
173 	 *           when the adaptor can't be instantiated, or when it can't be found
174 	 * @throws YADAResourceException
175 	 *           when a query's source attribute can't be found in the application
176 	 *           context, or there is another problem with the context
177 	 * @throws YADAAdaptorException
178 	 *           when a query cannot be built by the adaptor
179 	 * @throws YADARequestException
180 	 *           when filters are included in the request config, but can't be
181 	 *           converted into a JSONObject
182 	 * @throws YADAParserException
183 	 *           when query code cannot be parsed successfully
184 	 * @since 0.4.0.0
185 	 */
186 	private void processQueries() throws YADAQueryConfigurationException, YADAConnectionException, YADAFinderException, YADAResourceException, YADAUnsupportedAdaptorException, YADARequestException, YADAAdaptorException, YADAParserException
187 	{
188 		if (YADAUtils.hasJSONParams(this.yadaReq)) //TODO this is a redundant call, see the constructor
189 		{
190 			setQueries(endowQueries(getJsonParams()));
191 		} 
192 		else if (YADAUtils.hasQname(this.yadaReq))
193 		{
194 			setQueries(new YADAQuery[] {endowQuery(this.yadaReq.getQname())});
195 		} 
196 		else
197 		{
198 			String msg = "Your request must contain a 'qname', 'q', 'JSONParams', or 'j' parameter.";
199 			throw new YADARequestException(msg);
200 		}
201 		setGlobalHarmonyMaps();
202 		setQueryHarmonyMaps();
203 		prepQueriesForExecution();
204 	}

206 	/**
207 	 * Create a composite harmony map for all queries based on either {@link YADARequest#getHarmonyMap()} 
208 	 * or the harmony maps stored in each query via param {@link YADAQuery#getParam(String)} 
209 	 * named {@link YADARequest#PS_HARMONYMAP}.  This object, consistent in each request, enables inclusion
210 	 * of both mapped and unmapped columns in a single result, specifically necessary for 
211 	 * {@link YADARequest#FORMAT_CSV} and other delimited formats.
212 	 * @since 6.1.0
213 	 */
214 	private void setGlobalHarmonyMaps() 
215 	{
216 	  JSONArray  reqHm            = this.yadaReq.getHarmonyMap();
217 	  JSONObject globalHarmonyMap = new JSONObject();
218 	  if(null == reqHm)
219 	  {
220 	    for(YADAQuery yq : getQueries())
221 	    {
222 	      YADAParam p = yq.getParam(YADARequest.PS_HARMONYMAP);
223 	      if(p != null)
224 	      {
225 	        JSONObject j = new JSONObject(p.getValue());
226 	        if(j.length() > 0)
227 	        {
228 	          globalHarmonyMap = populateGlobalHarmonyMap(globalHarmonyMap, j);
229 	        }
230 	      }
231 	    }
232 	    for(YADAQuery yq : getQueries())
233 	      yq.setGlobalHarmonyMap(globalHarmonyMap);
234 	  }
235 	  else
236 	  {
237 	    for(int i=0;i<reqHm.length();i++)
238 	    {
239         JSONObject j = reqHm.getJSONObject(i);
240         if(j.length() > 0)
241         {
242           globalHarmonyMap = populateGlobalHarmonyMap(globalHarmonyMap, j);
243 	      }
244       }
245 	    for(YADAQuery yq : getQueries())
246 	      yq.setGlobalHarmonyMap(globalHarmonyMap);
247 	  }  
248 	}

250 	/**
251 	 * Populates a {@link JSONObject} with the unique set of keys, i.e., original column or field names 
252 	 * passed into the request in {@link JSONParams} or in the {@link YADARequest#PS_HARMONYMAP} parameter.
253 	 * 
254 	 * @param global The {@link JSONObject} to populate
255 	 * @param local The {@link JSONObject} containing the key/value pairs to transfer to {@code global}
256 	 * @return the {@code global} {@link JSONObject} containing the unique set of keys (and values)
257 	 * @since 6.1.0
258 	 */
259 	private JSONObject populateGlobalHarmonyMap(JSONObject global, JSONObject local)
260 	{
261 	  for(String key : JSONObject.getNames(local))
262     {
263       try
264       {
265         global.putOnce(key, local.get(key));
266       }
267       catch(JSONException e)
268       {
269         String msg = "Key ["+key+"] already exists in global harmony map.";
270         l.warn(msg);
271       }
272     }
273 	  return global;
274 	}

277 	 * Checks for a {@code harmonyMap} or {@code h} spec in the {@link #yadaReq}. If {@code null}, 
278 	 * iterates over the queries in the request, identifying those that have embedded {@code h} specs
279 	 * and those that do not. If any queries contain specs, those that do not are provided with empty maps.
280 	 * 
281 	 * @throws YADARequestException when the resulting harmonyMap is non-compliant
282 	 * @since 6.1.0
283 	 */
284 	private void setQueryHarmonyMaps() throws YADARequestException 
285 	{
286   	if(this.yadaReq.getHarmonyMap() == null)
287   	{
288   	  ArrayList<JSONObject> hasMap = new ArrayList<>();
289   	  ArrayList<YADAQuery> noMap  = new ArrayList<>();
290   	  for(YADAQuery yq : this.queries)
291   	  {
292   	    YADAParam p = yq.getParam(YADARequest.PS_HARMONYMAP);
293   	    if(p != null)
294   	    {
295     	    hasMap.add(new JSONObject(p.getValue()));
296   	    }
297   	    else
298   	    {
299   	      noMap.add(yq);
300   	    }
301   	  }
302   	  if(hasMap.size() > 0)
303   	  {
304   	    for(YADAQuery yq : noMap)
305   	    {
306   	      YADAParam param = new YADAParam(YADARequest.PS_HARMONYMAP,"{}",YADAParam.QUERY,YADAParam.OVERRIDEABLE);
307   	      yq.addParam(param);
308   	    }
309   	  }
310   	}
311 	}

314 	 * Adds the JDBC or SOAP connection object for {@code yq} to the internal
315 	 * index. If the connection is not yet in the index, it is created from the
316 	 * source stored in the {@code yq} object.  If the protocol value in the query
317 	 * is not SOAP or JDBC, the method exits silently
318 	 * 
319 	 * @param yq
320 	 *          the query object containing the source string for setting the
321 	 *          connection
322 	 * @throws YADAConnectionException
323 	 *           when the connection to the source in the query cannot be set
324 	 */
325 	private void storeConnection(YADAQuery yq) throws YADAConnectionException
326 	{
327 		String source = yq.getSource();
328 		if (!this.connectionMap.containsKey(source))
329 		{
330 			yq.setConnection();
331 			this.connectionMap.put(source, yq.getConnection());
332 		} 
333 		else
334 		{
335 			if (yq.getProtocol().equals(Parser.JDBC))
336 				yq.setConnection((Connection)this.connectionMap.get(source));
337 			else if (yq.getProtocol().equals(Parser.SOAP))
338 				yq.setSOAPConnection((SOAPConnection)this.connectionMap.get(source));
339 		}
340 	}

342 	/**
343 	 * Executes a query-level commit on the connection stored in the YADAQuery
344 	 * referenced by the parameter. If the connection object returned by the query
345 	 * is not a JDBC connection, but, for instance, a SOAPConnection, the error
346 	 * will be caught and handled gracefully.
347 	 * 
348 	 * @param yq
349 	 *          the query containing the statements to commit
350 	 * @throws YADAConnectionException
351 	 *           when the commit fails
352 	 */
353 	public void commit(YADAQuery yq) throws YADAConnectionException
354 	{
355 		try
356 		{
357 			if(this.requiredCommits.contains(yq.getSource()))
358 			{
359 				Connection connection = (Connection)yq.getConnection();
360 				if(connection.getHoldability() == ResultSet.HOLD_CURSORS_OVER_COMMIT)
361 				{
362 					connection.commit();
363 					int    count = yq.getResult().getTotalResultCount();
364 					String rows  = count == 1 ? "row" : "rows"; 
365 					String msg   = "\n------------------------------------------------------------\n";
366 					msg         += "   Commit successful on connection to ["+yq.getSource()+"] ("+count+" "+rows+")\n";
367 					msg         += "------------------------------------------------------------\n";
368 					l.debug(msg);
369 				}
370 				else
371 				{
372 					deferCommit(yq.getSource());
373 				}
374 			}
375 		} 
376 		catch (SQLException e)
377 		{
378 			String msg = "Unable to commit transaction on ["+yq.getSource()+"].";
379 			throw new YADAConnectionException(msg, e);
380 		} 
381 		catch (ClassCastException e)
382 		{
383 			l.info("Connection to ["+yq.getSource()+"] is not a JDBC connection (it's probably SOAP.) No commit was attempted.");
384 		}
385 	}

387 	/**
388 	 * Executes a commit on all connections created during processing of the
389 	 * current request.
390 	 * 
391 	 * @throws YADAConnectionException
392 	 *           when the commit fails
393 	 */
394 	public void commit() throws YADAConnectionException
395 	{
396 		if (this.connectionMap != null && this.connectionMap.keySet().size() > 0)
397 		{
398 			//TODO			int    totalCount = 0;
399 			String source     = "";
400 			for(Iterator<String> iterator = this.requiredCommits.iterator(); iterator.hasNext();)
401 			{
402 				try
403 				{
404 					source = iterator.next();
405 					Connection connection = (Connection)this.connectionMap.get(source);
406 					if(connection.getHoldability() == ResultSet.HOLD_CURSORS_OVER_COMMIT)
407 					{
408 						connection.commit();
409 						String msg = "\n------------------------------------------------------------\n";
410 						msg += "   Commit successful on ["+source+"].\n";
411 						msg += "------------------------------------------------------------\n";
412 						l.info(msg);
413 					}
414 					else
415 					{
416 						deferCommit(source);
417 					}
418 				}
419 				catch (SQLException e)
420 				{
421 					String msg = "Unable to commit transaction on ["+source+"].";
422 					throw new YADAConnectionException(msg, e);
423 				} 
424 				catch (ClassCastException e)
425 				{
426 					l.info("Connection to ["+source+"] is not a JDBC connection (it's probably SOAP.)  No commit was attempted.");
427 				} 
428 			}
429  		}
430 	}

445 	/**
446 	 * This method attempts (and usually suceeds) to close all JDBC resources
447 	 * opened during processing of the current request, including
448 	 * {@link ResultSet}s {@link java.sql.PreparedStatement}s and
449 	 * {@link java.sql.CallableStatement}s using the utility methods in
450 	 * {@link com.novartis.opensource.yada.ConnectionFactory}
451 	 * 
452 	 * @throws YADAConnectionException
453 	 *           when there is a problem closing any of the resources
454 	 * @see ConnectionFactory#releaseResources(ResultSet)
455 	 * @see ConnectionFactory#releaseResources(java.sql.Statement)
456 	 */
457 	public void releaseResources() throws YADAConnectionException
458 	{
459 		for (String source : this.deferredCommits)
460 		{
461 			try
462 			{
463 				Connection connection = (Connection)this.connectionMap.get(source);
464 				connection.commit();
465 				String msg = "\n------------------------------------------------------------\n";
466 				msg += "   Commit successful on ["+source+"].\n";
467 				msg += "------------------------------------------------------------\n";
468 				l.info(msg);
469 			}
470 			catch (SQLException e)
471 			{
472 				String msg = "\n------------------------------------------------------------\n";
473 				msg += "   Unable to commit transaction on ["+source+"].\n";
474 				msg += "------------------------------------------------------------\n";
475 				l.error(msg); //TODO should there be a rollback message here?
476 			} 
477 		}
478 		for (YADAQuery yq : this.getQueries())
479 		{
480 			YADAQueryResult yqr = yq.getResult();
481 			if (yqr != null && yqr.getResults() != null)
482 			{
483 				for (Object result : yqr.getResults())
484 				{
485 					if (result instanceof ResultSet)
486 					{
487 						ConnectionFactory.releaseResources((ResultSet)result);
488 					}
489 				}
490 			}
491 			if (yq.getPstmt() != null && yq.getPstmt().size() > 0)
492 			{
493 				for (PreparedStatement p : yq.getPstmt())
494 				{
495 					ConnectionFactory.releaseResources(p);
496 					this.pStmtMap.remove(p);
497 					l.debug("PreparedStatement removed from map.");
498 				}
499 			}
500 			if (yq.getPstmtForCount() != null && yq.getPstmtForCount().values()
501 																							.size() > 0)
502 			{
503 				for (PreparedStatement p : yq.getPstmtForCount().values())
504 				{
505 					ConnectionFactory.releaseResources(p);
506 					this.pStmtMap.remove(p);
507 					l.debug("PreparedStatement for Count removed from map.");
508 				}
509 			}
510 			if (yq.getCstmt() != null && yq.getCstmt().size() > 0)
511 			{
512 				for (CallableStatement c : yq.getCstmt())
513 				{
514 					ConnectionFactory.releaseResources(c);
515 					this.cStmtMap.remove(c);
516 					l.debug("CallableStatement removed from map.");
517 				}
518 			}
519 		}
520 		this.connectionMap.clear();
521 		l.debug("QueryManager connection map has been cleared.");
522 	}

657 	private void prepQueriesForExecution() throws YADAResourceException, YADAUnsupportedAdaptorException, YADAConnectionException, YADARequestException, YADAAdaptorException, YADAParserException
658 	{
659 
660 		for (YADAQuery yq : this.getQueries())
661 		{
662 			String source        = yq.getSource();
663 			String conformedCode = yq.getConformedCode();
664 			String wrappedCode   = "";
665 			int    dataSize      = yq.getData().size() > 0 ? yq.getData().size() : 1;
666 			if(this.qutils.requiresConnection(yq))
667 			{	
668 				storeConnection(yq);
669 			}
670 			
671 			this.qutils.processStatement(yq);
672 			
673 			if(yq.getProtocol().equals(Parser.JDBC))
674 			{
675 				if(yq.getType().equals(Parser.CALL))
676 				{
677 					for (int row = 0; row < dataSize; row++)
678 					{
679 						wrappedCode = ((JDBCAdaptor)yq.getAdaptor()).buildCall(conformedCode).toString();
680 						String msg  = "\n------------------------------------------------------------";
681 						       msg += "\n   Callable statement to execute:";
682 						       msg += "\n------------------------------------------------------------\n";
683 						       msg += wrappedCode.toString() + "\n";
684 						l.debug(msg);
685 						storeCallableStatement(yq, wrappedCode);
686 					}
687 					this.requiredCommits.add(yq.getSource());
688 				}
689 				else
690 				{
691 					boolean    count     = Boolean.valueOf(yq.getYADAQueryParamValue(YADARequest.PS_COUNT)[0]).booleanValue();
692 					boolean    countOnly = Boolean.valueOf(yq.getYADAQueryParamValue(YADARequest.PS_COUNTONLY)[0]).booleanValue();
693 					int        pageStart = Integer.valueOf(yq.getYADAQueryParamValue(YADARequest.PS_PAGESTART)[0]).intValue();
694 					int        pageSize  = Integer.valueOf(yq.getYADAQueryParamValue(YADARequest.PS_PAGESIZE)[0]).intValue();
695 					if (pageSize == -1)
696 						pageSize = YADAUtils.ONE_BILLION;
697 					int        firstRow  = 1 + (pageStart * pageSize) - pageSize;
698 					String     sortOrder = yq.getYADAQueryParamValue(YADARequest.PS_SORTORDER)[0];
699 					String     sortKey   = "";
700 					JSONObject filters   = null;
701 					
702 					if (yq.hasParamValue(YADARequest.PS_SORTKEY))
703 					{
704 						sortKey = yq.getYADAQueryParamValue(YADARequest.PS_SORTKEY)[0];
705 					}
706 					
707 					try
708 					{
709 						if (yq.hasParamValue(YADARequest.PS_FILTERS))
710 						{
711 							filters = new JSONObject(yq.getYADAQueryParamValue(YADARequest.PS_FILTERS)[0]);
712 						}
713 					} 
714 					catch (JSONException e)
715 					{
716 						String msg = "Error while getting filters from parameters.";
717 						throw new YADARequestException(msg, e);
718 					}
719 					
720 					for (int row = 0; row < dataSize; row++)
721 					{
722 						if(yq.getIns() != null && yq.getIns().length > 0)
723 						{
724 							conformedCode = this.qutils.getConformedCode(this.qutils.processInColumns(	yq, row));
725 						}
726 						
727 						if(yq.getType().equals(Parser.SELECT))
728 						{
729 							wrappedCode = ((JDBCAdaptor)yq.getAdaptor()).buildSelect(	conformedCode,
730 																																				sortKey,
731 																																				sortOrder,
732 																																				firstRow,
733 																																				pageSize,
734 																																				filters).toString();
735 							String msg  = "\n------------------------------------------------------------";
736 										 msg += "\n   SELECT statement to execute:";
737 										 msg += "\n------------------------------------------------------------\n";
738 										 msg += wrappedCode.toString() + "\n";
739 							l.debug(msg);
740 						}
741 						else // INSERT, UPDATE, DELETE
742 						{
743 							wrappedCode = conformedCode;
744 							this.requiredCommits.add(yq.getSource());
745 							String msg  = "\n------------------------------------------------------------";
746 							       msg += "\n   INSERT/UPDATE/DELETE statement to execute:";
747 							       msg += "\n------------------------------------------------------------\n";
748 							       msg += wrappedCode.toString() + "\n";
749 							l.debug(msg);
750 						}
751 						
752 						storePreparedStatement(yq, wrappedCode);
753 						if (yq.getType().equals(Parser.SELECT) && (count || countOnly))
754 						{
755 							wrappedCode = ((JDBCAdaptor)yq.getAdaptor()).buildSelectCount(conformedCode,filters).toString();
756 						  String msg  = "\n------------------------------------------------------------";
757 							       msg += "\n   SELECT COUNT statement to execute:";
758 							       msg += "\n------------------------------------------------------------\n";
759 							       msg += wrappedCode.toString() + "\n";
760 							l.debug(msg);
761 							storePreparedStatementForCount(yq, yq.getPstmt(row), wrappedCode);
762 						}
763 						this.qutils.setValsInPosition(yq, row);
764 					} // end data loop
765 				} // end if callable
766 			} // end if JDBC
767 			else if(yq.getProtocol().equals(Parser.SOAP)
768 					|| yq.getProtocol().equals(Parser.REST)
769 					|| yq.getProtocol().equals(Parser.FILE))
770 			{
771 				if(yq.getType().equals(Parser.SOAP))
772 				{
773 					yq.setSource(source.replace(SOAPAdaptor.PROTOCOL_SOAP,SOAPAdaptor.PROTOCOL_HTTP));
774 					for (int row = 0; row < dataSize; row++)
775 					{
776 						wrappedCode = ((SOAPAdaptor)yq.getAdaptor()).build(yq);
777 						storeSoapMessage(yq, wrappedCode);
778 						this.qutils.setValsInPosition(yq, row);
779 					}
780 				}
781 				else if(yq.getType().equals(Parser.REST))
782 				{
783 					for (int row = 0; row < dataSize; row++)
784 					{
785 						wrappedCode = ((RESTAdaptor)yq.getAdaptor()).build(yq);
786 						storeRestQuery(yq, wrappedCode);
787 						this.qutils.setValsInPosition(yq, row);
788 					}
789 				}
790 				else // filesystem
791 				{
792 					for (int row = 0; row < dataSize; row++)
793 					{
794 						wrappedCode = ((FileSystemAdaptor)yq.getAdaptor()).build(yq);
795 						storeRestQuery(yq, wrappedCode);
796 						this.qutils.setValsInPosition(yq, row);
797 					}
798 				}
799 			}
800 			else
801 			{
802 				String msg = "The query you are attempting to execute requires ";
803 				msg += "a protocol or class that is not supported.  This ";
804 				msg += "could be the result of a configuration issue.";
805 				throw new YADAUnsupportedAdaptorException(msg);
806 			} // end protocols
807 		} // query loop
808 	} // end prepareQueryPackages
